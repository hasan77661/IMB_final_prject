/*1.Sales performance by product, store, and city*/
SELECT 
  p.product_id,
  c.store_id,
  c.city_id,
  SUM (sl.sales) AS total_sales,
  SUM (sl.revenue) AS total_revenue,
  AVG (sl.price) AS average_price
FROM 
  sales sl
INNER JOIN product_hierarchy p ON sl.product_id = p.product_id
INNER JOIN store_cities c ON sl.store_id = c.store_id
GROUP BY p.product_id, c.store_id, c.city_id;


/*2. Sales trends over time for each store and accumulate grand totals(ROLLUP)*/
SELECT 
    store_id,
    TO_CHAR(date, 'YYYY-MM') AS sales_month,
    SUM(sales) AS total_sales
FROM 
    sales
GROUP BY 
    ROLLUP (store_id, sales_month);


/*3. A rolled-up summary of sales data across product hierarchy levels(ROLLUP)*/
SELECT 
    p.hierarchy1_id,
    p.hierarchy2_id,
    SUM(s.sales) AS total_sales
FROM 
    sales s
JOIN 
    product_hierarchy p 
ON 
    s.product_id = p.product_id
GROUP BY 
    ROLLUP (p.hierarchy1_id, p.hierarchy2_id);

/*3.Sales for Each Month and Each Store(GROUPING SETS)*/
SELECT 
    store_id,
    TO_CHAR(date, 'YYYY-MM') AS sales_month,
    SUM(sales) AS total_sales
FROM 
    sales
GROUP BY 
     GROUPING SETS(store_id, sales_month);


/*4.Combine sales data with geographical information to observe sales trends over time and across various regions(CUBE)*/
SELECT 
    c.city_id, 
    TO_CHAR(s.date, 'YYYY-MM') AS sales_month,
    SUM(s.sales) AS total_sales
FROM 
    sales s
JOIN 
    store_cities c 
ON 
    s.store_id = c.store_id
GROUP BY 
    CUBE (c.city_id, sales_month);
